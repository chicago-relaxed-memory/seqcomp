\begin{definition}
  We say $\labelingAct(\bEv)=(\DW[]{x}{v})$ \emph{fulfills}
  $\labelingAct(\aEv)=(\DR[]{x}{v})$ if
  \begin{description}
  \item[{\labeltextsc[F3]{(F3)}{rf3}}] $\bEv \le \aEv$ and
    $\bEv \ltstrong \aEv$
  \item[{\labeltextsc[F4]{(F4)}{rf4}}]
    $\forall\labelingAct(\cEv)=(\DW[]{x}{..})$ either $\cEv \le \bEv$ or
    $\aEv \le \cEv$
  \end{description}  
\end{definition}
