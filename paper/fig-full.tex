$(\exists\cVal:\aEvs\fun\Val)$

\item[L2)] $\labelingAct(\aEv) = \DRREFP{\cVal_\aEv}{\aVal_\aEv}$,
\item[L3)] $\labelingForm(\aEv)$ implies
  \begin{math}
    \bForm_\aEv
    \land \cExp{=}\cVal_\aEv
    \land \RO
    \land \QL{\REF{\cVal_\aEv}}{\amode}
  \end{math},
  % where    
  % $\QL{}{\mSC}=\Q{\mSC}$ and otherwise $\QL{}{\amode}=\QwREF{\cVal_\aEv}$, % for $\amode\neq\mRLX$,
\item[L4)]
  \begin{math}
    (\forall\dVal)
  \end{math}
  if $\aEv\in\aEvs\cap\bEvs$ then
  $\aTr[\bEvs](\aForm)$ implies
  \begin{math}
    \bForm_\aEv
    \limplies (\cExp{=}\dVal)
    \limplies (\aVal{=}\uReg{\aEv})
    \limplies \aForm[\uReg{\aEv}/\aReg]%[\uReg{\aEv}/\REF{\dVal}]
  \end{math},
  \makebox[4.8cm]{}
\item[L5)] 
  \begin{math}
    (\forall\dVal)
  \end{math}
  if $\aEv\in\aEvs\setminus\bEvs$ then
  $\aTr[\bEvs](\aForm)$ implies
  \begin{math}
    \bForm_\aEv
    \limplies (\cExp{=}\dVal)
    \limplies \PBRbig{        
      \DL{\REF{\dVal}}{\amode}
      \land \lnot\QxREF{\dVal}
      \land
      (\RW
      \limplies (\aVal{=}\uReg{\aEv}\lor\aLoc{=}\uReg{\aEv}) 
      \limplies \aForm[\uReg{\aEv}/\aReg]%[\uReg{\aEv}/\REF{\dVal}]
      )
    }      
  \end{math},
\item[L6)] % if 
  % \begin{math}
  %   (\forall\aEv\in\bEvs)(\cForm \textimplies
  %   \lnot\bForm_\aEv)
  % \end{math}
  % then

    (\forall\dVal)
    (\forall\bReg)
  \end{math}
  $\aTr[\bEvs](\aForm)$ implies 
  \begin{math}
    (\not\exists\aEv\in\aEvs \suchthat \bForm_\aEv)
    \limplies (\cExp{=}\dVal)
    \limplies \PBR{        
      \DL{\REF{\dVal}}{\amode}
      \land \lnot\QxREF{\dVal}
      \land
      \limplies \aForm[\bReg/\aReg]%[\bReg/\REF{\dVal}]
    }.
  \end{math}
